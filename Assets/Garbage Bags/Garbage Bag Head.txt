local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()

local function attachGarbageBagToHead()
	local head = character:FindFirstChild("Head")
	if not head then return end

	for _, tool in ipairs(character:GetChildren()) do
		if tool:IsA("Tool") and tool.Name == "Garbage Bag" and tool:FindFirstChild("Handle") then
			tool.Grip = CFrame.new(Vector3.new(1.45, -2, -1.5)) -- Adjust to fit inside the head
			tool.Parent = character
		end
	end
end

RunService.RenderStepped:Connect(function()
	pcall(attachGarbageBagToHead)
end)
